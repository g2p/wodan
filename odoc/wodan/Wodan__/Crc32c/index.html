Module Wodan__.Crc32c

val cstruct_valid : Cstruct.t -> bool

cstruct_valid cstruct returns whether the CRC32C of cstruct is 0xffffffffl

val cstruct_reset : Cstruct.t -> unit

cstruct_reset cstruct rewrites the last four bytes of cstruct so that cstruct_valid cstruct is true
